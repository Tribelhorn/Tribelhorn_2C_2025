64void convertToBcdArray (uint32_t data, uint8_t digits, uint8_t * bcd_number) {
65
66 for (int i = (digits-1); i > -1; i--) { // Recorre la el arreglo desde el final
67 uint32_t dato_sobre_10 = data / 10; // Divide el número por 10
68 uint32_t dato_entero = dato_sobre_10 * 10; // Obtiene la parte entera multiplicando por 10 para truncarlo
69 uint8_t digito = data - dato_entero; // Se hace la diferencia para obtener el dígito menos significativo
70 *(bcd_number + i) = digito; // Guarda el dígito en el arreglo
71 data = dato_sobre_10; // Cambia el dato para la siguiente iteración

98void escribirGPIO(uint8_t bcd, gpioConf_t* gpios) {
99
100 for (int i = 0; i < 4; i++) {
101 GPIOInit(gpios[i].pin, gpios[i].dir); // Inicializa el pin
102
103 if (bcd & (1 << i)) { // Recorre cada bit del bcd preguntando si está en 1
104 GPIOOn(gpios[i].pin); // Enciende el pin
105 } else { // Si el bit i está en 0
106 GPIOOff(gpios[i].pin); // Apaga el pin
